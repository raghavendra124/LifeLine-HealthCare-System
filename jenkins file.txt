pipeline {
    agent 'any'
    tools {
        jdk 'jdk1.8'
        maven 'maven3'
    }

    stages{
        stage('git pull'){
            steps{
                echo 'code pull from https://github.com/raghavendra124/LifeLine-HealthCare-System.git'
                git 'https://github.com/raghavendra124/LifeLine-HealthCare-System.git'
            }
        }
      
        stage('Maven Build'){
            steps{
                echo 'Maven clean and build'
                sh 'mvn clean'
                sh 'mvn install'
            }
        }
       
        stage('docker image'){
            steps{
                echo 'docker image build, push to ecr'
                sh 'docker -v'
                sh 'docker build -t app:1 .'
                sh '$(aws ecr get-login --no-include-email)'
                sh 'docker tag app:1 docker push raghavendrachari14/application1:latest'
                sh 'docker push docker push raghavendrachari14/application1:latest'
            }
        
        }

}
